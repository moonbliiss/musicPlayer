?? Реализация без роутера
1. Главный компонент App с состоянием текущего экрана
tsx
// App.tsx
import { useState } from 'react';
import Sidebar from './components/Sidebar';
import Player from './components/Player';
import AllSongs from './pages/AllSongs';
import Favorites from './pages/Favorites';
import PlaylistDetail from './pages/PlaylistDetail';
import { MusicProvider } from './context/MusicContext';

type View = 
  | { type: 'allMusic' }
  | { type: 'favoriteMusic' }
  | { type: 'playlist'; playlistId: string };

const App = () => {
  const [currentView, setCurrentView] = useState<View>({ type: 'allMusic' });

  // Рендерим нужную страницу
  const renderContent = () => {
    switch (currentView.type) {
      case 'allMusic':
        return <AllSongs />;
      case 'favoriteMusic':
        return <Favorites />;
      case 'playlist':
        return <PlaylistDetail playlistId={currentView.playlistId} />;
    }
  };

  return (
    <MusicProvider>
      <div className="app-layout">
        <Sidebar currentView={currentView} onNavigate={setCurrentView} />
        <main className="content">
          {renderContent()}
        </main>
        <Player />
      </div>
    </MusicProvider>
  );
};





2. Компонент Sidebar (управляет навигацией)
tsx
// components/Sidebar.tsx
import { useMusic } from '../context/MusicContext';

interface SidebarProps {
  currentView: View;
  onNavigate: (view: View) => void;
}

const Sidebar = ({ currentView, onNavigate }: SidebarProps) => {
  const { playlists, addPlaylist } = useMusic();

  const handleAddPlaylist = () => {
    const name = prompt('Название плейлиста');
    if (name) {
      const newPlaylist = addPlaylist(name);
      onNavigate({ type: 'playlist', playlistId: newPlaylist.id });
    }
  };

  const isActive = (view: View) => {
    if (view.type === 'allMusic' && currentView.type === 'allMusic') return true;
    if (view.type === 'favoriteMusic' && currentView.type === 'favoriteMusic') return true;
    if (
      view.type === 'playlist' && 
      currentView.type === 'playlist' && 
      view.playlistId === currentView.playlistId
    ) return true;
    return false;
  };

  return (
    <aside>
      <nav>
        <div 
          onClick={() => onNavigate({ type: 'allMusic' })}
          className={isActive({ type: 'allMusic' }) ? 'active' : ''}
        >
          ?? Все треки
        </div>
        <div 
          onClick={() => onNavigate({ type: 'favoriteMusic' })}
          className={isActive({ type: 'favoriteMusic' }) ? 'active' : ''}
        >
          ?? Избранные
        </div>
        
        <hr />
        <h4>Плейлисты</h4>
        {playlists.map(pl => (
          <div
            key={pl.id}
            onClick={() => onNavigate({ type: 'playlist', playlistId: pl.id })}
            className={isActive({ type: 'playlist', playlistId: pl.id }) ? 'active' : ''}
          >
            ?? {pl.name}
          </div>
        ))}
        
        <button onClick={handleAddPlaylist}>+ Создать плейлист</button>
      </nav>
    </aside>
  );
};





3. Страницы (получают данные из контекста)
tsx
// pages/AllSongs.tsx
import { useMusic } from '../context/MusicContext';
import SongList from '../components/SongList';

const AllSongs = () => {
  const { songs } = useMusic();
  return <SongList songs={songs} title="Все треки" />;
};

// pages/Favorites.tsx
const Favorites = () => {
  const { favorites } = useMusic();
  return <SongList songs={favorites} title="Избранные" />;
};

// pages/PlaylistDetail.tsx
import { useMusic } from '../context/MusicContext';

const PlaylistDetail = ({ playlistId }: { playlistId: string }) => {
  const { playlists } = useMusic();
  const playlist = playlists.find(p => p.id === playlistId);
  
  if (!playlist) return <div>Плейлист не найден</div>;
  return <SongList songs={playlist.songs} title={playlist.name} />;
};





4. Контекст для данных (MusicContext)
tsx
// context/MusicContext.tsx
import { createContext, useContext, useState, ReactNode } from 'react';

interface Song {
  id: string;
  title: string;
  artist: string;
  url: string;
}

interface Playlist {
  id: string;
  name: string;
  songs: Song[];
}

interface MusicContextType {
  playlists: Playlist[];
  songs: Song[];
  favorites: Song[];
  addPlaylist: (name: string) => Playlist;
  toggleFavorite: (song: Song) => void;
}

const MusicContext = createContext<MusicContextType | undefined>(undefined);

export const MusicProvider = ({ children }: { children: ReactNode }) => {
  const [playlists, setPlaylists] = useState<Playlist[]>([]);
  const [songs, setSongs] = useState<Song[]>([
    { id: '1', title: 'Song 1', artist: 'Artist A', url: '/audio/1.mp3' },
    // ... начальные песни
  ]);
  const [favorites, setFavorites] = useState<Song[]>([]);

  const addPlaylist = (name: string) => {
    const newPlaylist = { id: Date.now().toString(), name, songs: [] };
    setPlaylists(prev => [...prev, newPlaylist]);
    return newPlaylist;
  };

  const toggleFavorite = (song: Song) => {
    setFavorites(prev => 
      prev.some(s => s.id === song.id)
        ? prev.filter(s => s.id !== song.id)
        : [...prev, song]
    );
  };

  return (
    <MusicContext.Provider value={{ playlists, songs, favorites, addPlaylist, toggleFavorite }}>
      {children}
    </MusicContext.Provider>
  );
};

export const useMusic = () => {
  const context = useContext(MusicContext);
  if (!context) throw new Error('useMusic must be used within MusicProvider');
  return context;
};

?? Преимущества этого подхода
Нет лишних зависимостей — не нужен react-router-dom.

Проще код — никаких <Routes>, <Outlet>, useParams, useNavigate.

Быстрее — нет маршрутизации, просто условный рендеринг.

Понятнее — вся логика навигации в одном месте (Sidebar передает onNavigate, App рендерит контент).

?? Когда нужно добавить роутер?
Вернитесь к React Router, если:

Понадобятся прямые ссылки на плейлист (например, /playlist/123).

Нужна история браузера (кнопки «Назад/Вперед»).

Нужно обновлять страницу и оставаться на том же месте.

Если ничего из этого не нужно — оставляйте как есть. Вы получили легкое, быстрое и понятное приложение без лишнего оверинжиниринга. ??
